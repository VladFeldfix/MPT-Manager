//TEST DIMMER
PrintLn(4,"TEST DIMMER DIMNAME - 2 Wire")
Prompt("Set dimmer DIMNAME max clockwise and press Continue")
SetResistance(LV, Pass = MAXOHM Ohms +- 5%, I = Auto)
Resistor(POINT3, POINT2)
SetConductor(HC, Pass < MINOHM Ohm, I = 1000 mA, V = 5 Volts)
Continuity(POINT3, POINT1)
Prompt("Set dimmer DIMNAME max counter-clockwise and press Continue")
SetResistance(LV, Pass = MAXOHM Ohms +- 5%, I = Auto)
Resistor(POINT3, POINT1)
SetConductor(HC, Pass < MINOHM Ohm, I = 1000 mA, V = 5 Volts)
Continuity(POINT3, POINT2)
Prompt("Set dimmer DIMNAME to neutral position")
WaitForNoCont(POINT3, POINT1)
WaitForNoCont(POINT3, POINT2)
